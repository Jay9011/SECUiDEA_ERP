71 public async Task<TokenResponse> RefreshTokenAsync(string refreshToken, string ipAddress)
72 {
73 // RefreshToken 조회
74 RefreshToken token = await _tokenRepository.GetByTokenAsync(refreshToken);
75 if (token == null || !token.IsActive)
76 {
77 return null;
78 }
79
80 // 세션 유효성 확인
81 bool isValidSession = await _sessionService.ValidateSessionAsync(token.UserId, token.Provider, token.SessionId);
82
83 if (!isValidSession)
84 {
85 await RevokeTokenAsync(refreshToken, ipAddress, "Session is no longer valid");
86 return null;
87 }
88
89 // 사용자 정보 조회
90 User user = await GetUserByIdAsync(token.Provider, token.UserId);
91 if (user == null)
92 {
93 return null;
94 }
95
96 // 새로운 JWT 토큰 생성
97 string newAccessToken = _jwtService.GenerateJwtToken(user, token.SessionId);
98
99 // 세션 활동 시간 업데이트
100 await _sessionService.UpdateSessionActivityAsync(token.Provider, token.UserId, token.SessionId);
101
102 return new TokenResponse
103 {
104 AccessToken = newAccessToken,
105 AccessExpiryDate = DateTime.Now.AddMinutes(_jwtService.GetSettings.ExpiryMinutes),
106 RefreshToken = token.Token,
107 RefreshExpiryDate = token.ExpiryDate,
108 EnableSessionTimeout = user.EnableSessionTimeout,
109 SessionId = token.SessionId,
110 SessionExpiryDate = DateTime.Now.AddMinutes(user.SessionTimeoutMinutes)
111 };
112 }

141 private async Task RevokeTokenAsync(string token, string ipAddress, string reason)
142 {
