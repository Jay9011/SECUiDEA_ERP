128 return BadRequest("Invalid visit date format.");
129 }
130

132

143 VisitEDate = visitEndDate,

159 visitReserveVisitantId = dr.Field<int>("VisitReserveVisitantID");

164 { "visitantId", visitantId },
165 { "visitReserveId", visitReserveId },
166 { "visitReserveVisitantId", visitReserveVisitantId },
