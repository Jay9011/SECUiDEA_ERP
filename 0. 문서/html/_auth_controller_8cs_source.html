23 [HttpPost]
24 public async Task<IActionResult> RefreshToken()
25 {
26 // 우선, 쿠키에서 토큰 확인
27 string? refreshToken = Request.Cookies[StringClass.RefreshToken];
28
29 // 쿠키에 없으면 요청 헤더나 본문에서 확인
30 if (string.IsNullOrEmpty(refreshToken) && Request.Headers.TryGetValue(StringClass.HXRefreshToken, out var headerToken))
31 {
32 refreshToken = headerToken;
33 }
34 else if (string.IsNullOrEmpty(refreshToken))
35 {
36 var requestBody = await GetRequestBodyAsync();
37 if (requestBody != null && requestBody.TryGetValue(StringClass.RefreshToken, out var bodyToken))
38 {
39 refreshToken = bodyToken?.ToString();
40 }
41 }
42
43 if (string.IsNullOrEmpty(refreshToken))
44 {
45 return Unauthorized(new { message = "Refresh token is required" });
46 }
47
48 string ipAddress = GetClientIpAddress();
49 var tokenResponse = await _authService.RefreshTokenAsync(refreshToken, ipAddress);
50
51 if (tokenResponse == null)
52 {
53 return Unauthorized(new { message = "Invalid refresh token" });
54 }
55
56 return Ok(tokenResponse);
57 }
