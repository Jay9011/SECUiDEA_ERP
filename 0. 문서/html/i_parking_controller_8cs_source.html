391 private bool GetDecryptedApiCredentials(out string? id, out string? password, out string? destinationId)
392 {
393 id = null;
394 password = null;
395 destinationId = null;
396
397 // API Setup에서 암호화된 ID, 비밀번호, 목적지 ID를 가져옴
398 try
399 {
400 id = _apiSetup.GetConnectionInfo().GetEndpoint(EncryptedId);
401 id = _crypto.Decrypt(id);
402
403 password = _apiSetup.GetConnectionInfo().GetEndpoint(EncryptedPwd);
404 password = _crypto.Decrypt(password);
405
406 destinationId = _apiSetup.GetConnectionInfo().GetEndpoint(EncryptedDestinationId);
407 destinationId = _crypto.Decrypt(destinationId);
408
409 return true;
410 }
411 catch (Exception e)
412 {
413 return false;
414 }
415
416 return false;
417 }

446 private void DecryptEndpoints(Dictionary<string, string> encryptedEndpoints)
447 {
448 foreach (var (key, value) in encryptedEndpoints)
449 {
450 if (string.Equals(key, EncryptedId)
451 || string.Equals(key, EncryptedPwd)
452 || string.Equals(key, EncryptedDestinationId))
453 {
454 encryptedEndpoints[key] = _crypto.Decrypt(value);
455 }
456 }
457 }
458
463 private void EncryptEndpoints(Dictionary<string, string> plainEndpoints)
464 {
465 foreach (var (key, value) in plainEndpoints)
466 {
467 if (string.Equals(key, EncryptedId)
468 || string.Equals(key, EncryptedPwd)
469 || string.Equals(key, EncryptedDestinationId))
470 {
471 plainEndpoints[key] = _crypto.Encrypt(value);
472 }
473 }
