62 public string GenerateJwtToken(User user, string? sessionId = null)
63 {
64 var tokenHandler = new JwtSecurityTokenHandler();
65 var key = Convert.FromBase64String(_jwtSettings.Secret);
66
67 var claims = new List<Claim>
68 {
69 new Claim(ClaimTypes.NameIdentifier, user.ID), // 사용자 ID 정보
70 new Claim(ClaimTypes.Name, user.Name), // 사용자 이름 정보
71 new Claim(ClaimTypes.Sid, user.Seq), // 사용자 시퀀스 정보
72 new Claim(ClaimTypes.Role, user.UserRole) // 사용자 역할 정보
73 };
74
75 // 세션 ID 추가
76 if (!string.IsNullOrEmpty(sessionId))
77 {
78 claims.Add(new Claim(StringClass.SessionId, sessionId));
79 }
80
81 #region 권한 정보 추가
82
83 // 사용자 권한 추가 (권한 정보는 List 형식으로 가져와야 함)
84 // 값은 Feature:PermissionLevel 형식으로 Claim에 추가
85 foreach (var permission in user.Permissions)
86 {
87 claims.Add(new Claim($"{StringClass.Permission}:{permission.Feature}", permission.Level.ToString()));
88 }
89
90 #endregion
91
92 // 토큰 생성 및 서명
93 var tokenDescriptor = new SecurityTokenDescriptor
94 {
95 Subject = new ClaimsIdentity(claims), // 모든 Claim 정보 추가
96 Issuer = _jwtSettings.Issuer, // 토큰 발급자 정보
97 Audience = _jwtSettings.Audience, // 토큰 대상자 정보
98 Expires = DateTime.Now.AddMinutes(_jwtSettings.ExpiryMinutes), // 토큰 만료 시간 정보
99 SigningCredentials = new SigningCredentials(
100 new SymmetricSecurityKey(key), // 토큰 서명 대칭키
101 SecurityAlgorithms.HmacSha256Signature // 토큰 서명 해시 알고리즘
102 )
103 };
104
105 var token = tokenHandler.CreateToken(tokenDescriptor);
106 return tokenHandler.WriteToken(token);
107 }

164 public RefreshToken GenerateRefreshToken(string userId, string ipAddress, string sessionId, bool rememberMe = false)
165 {
166 using var rng = RandomNumberGenerator.Create();
167 var randomBytes = new byte[64];
168 rng.GetBytes(randomBytes);
169
170 var tokenValue = Convert.ToBase64String(randomBytes);
171
172 var expiryDate = rememberMe
173 ? DateTime.Now.AddDays(_jwtSettings.AutoLoginDays)
174 : DateTime.Now.AddDays(_jwtSettings.RefreshTokenDays);
175
176 return new RefreshToken
177 {
178 UserId = userId,
179 Token = tokenValue,
180 ExpiryDate = expiryDate,
181 CreatedByIp = ipAddress,
182 CreatedAt = DateTime.Now,
183 SessionId = sessionId,
184 LastActivityDate = DateTime.Now
185 };
186 }

194 public bool ValidateToken(string token, out ClaimsPrincipal principal)
195 {
196 principal = null;
197
198 try
199 {
200 var tokenHandler = new JwtSecurityTokenHandler();
201 var key = Convert.FromBase64String(_jwtSettings.Secret);
202
203 var validationParameters = new TokenValidationParameters
204 {
205 ValidateIssuer = true,
206 ValidateAudience = true,
207 ValidateLifetime = true,
208 ValidateIssuerSigningKey = true,
209 ValidIssuer = _jwtSettings.Issuer,
210 ValidAudience = _jwtSettings.Audience,
211 IssuerSigningKey = new SymmetricSecurityKey(key),
212 ClockSkew = TimeSpan.Zero
213 };
214
215 principal = tokenHandler.ValidateToken(token, validationParameters, out _);
216 return true;
217 }
218 catch (Exception e)
219 {
220 return false;
221 }
222 }

264 public static void JwtConfigSetup(IIOHelper regIoHelper, ICryptoManager cryptoManager)
265 {
266 try
267 {
268 // 이미 설정이 있는 경우 바로 종료
269 {
270 var existingSettings = TryGetJwtCollection(regIoHelper, cryptoManager);
271 if (existingSettings.Count > 0
272 && existingSettings.ContainsKey(StringClass.Secret)
273 && existingSettings.TryGetValue(StringClass.Issuer, out string? issuer))
274 {
275 if (issuer == StringClass.SECUiDEA_Issuer)
276 {
277 return;
278 }
279 }
280 }
281
282 var jwtSettings = new Dictionary<string, string>();
283
284 string secret = GenerateSecureKey(64);
285 SetJwtCollection(jwtSettings, secret, cryptoManager);
286 regIoHelper.WriteSection(StringClass.JwtSettings, jwtSettings);
287
288 // 설정을 저장한 후, 다시 읽어와서 확인
289 if (cryptoManager.Decrypt(regIoHelper.ReadValue(StringClass.JwtSettings, StringClass.Secret))
290 != secret)
291 {
292
293 throw new Exception("Failed to save the secret key.");
294 }
295 }
296 catch (Exception e)
297 {
298 throw new Exception("Failed to save the secret key.");
299 }
300 }

362 private static Dictionary<string, string> GetJwtDictionary(IIOHelper regIoHelper, ICryptoManager? cryptoManager = null)
363 {
364 var jwtSettings = new Dictionary<string, string>();
365
366 var existingSettings = regIoHelper.ReadSection(StringClass.JwtSettings);
367 if (existingSettings is { Count: > 0 })
368 {
369 foreach (var key in existingSettings.Keys)
370 {
371 jwtSettings[key] = existingSettings[key];
372 }
373 }
374
375 if (cryptoManager != null)
376 {
377 foreach (var key in _cryptKeys)
378 {
379 if (jwtSettings.ContainsKey(key))
380 {
381 jwtSettings[key] = cryptoManager.Decrypt(jwtSettings[key]);
382 }
383 }
384 }
385
386 return jwtSettings;
387 }

393 private JwtSettings LoadJwtSettings()
394 {
395 var values = GetJwtCollection(_registryHelper, _cryptoManager);
396
397 var settings = new JwtSettings
398 {
399 Secret = values[StringClass.Secret],
400 Issuer = values[StringClass.Issuer],
401 Audience = values[StringClass.Audience],
402 };
403
404 // JWT 만료 시간 설정
405 if (double.TryParse(values[StringClass.ExpiryMinutes], out var expiryMinutes))
406 {
407 settings.ExpiryMinutes = expiryMinutes;
408 }
409
410 // Refresh Token 만료 시간 설정
411 if (double.TryParse(values[StringClass.RefreshTokenDays], out var refreshTokenDays))
412 {
413 settings.RefreshTokenDays = refreshTokenDays;
414 }
415
416 // 자동 로그인 만료 시간 설정
417 if (double.TryParse(values[StringClass.AutoLoginDays], out var autoLoginDays))
418 {
419 settings.AutoLoginDays = autoLoginDays;
420 }
421
422 // 중복 로그인 방지 설정
423 if (bool.TryParse(values[StringClass.PreventDuplicateLogin], out var preventDuplicateLogin))
424 {
425 settings.PreventDuplicateLogin = preventDuplicateLogin;
426 }
427
428 // 비활성 타임아웃 설정
429 if (bool.TryParse(values[StringClass.InactivityTimeout], out var inactivityTimeout))
430 {
431 settings.InactivityTimeout = inactivityTimeout;
432 }
433
434 // 비활성 타임아웃 시간 설정
435 if (double.TryParse(values[StringClass.InactivityTimeoutMinutes], out var inactivityTimeoutMinutes))
436 {
437 settings.InactivityTimeoutMinutes = inactivityTimeoutMinutes;
438 }
439
440 return settings;
441 }
