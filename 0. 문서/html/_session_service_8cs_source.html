78 public async Task<string> CreateSessionAsync(string provider, string userId, string ipAddress, bool rememberMe = false)
79 {
80 string sessionId = Guid.NewGuid().ToString();
81 string userKey = GetUserKey(userId, provider);
82 string userSessionKey = GetUserSessionKey(userId, provider);
83
84 // DB에 세션 정보 저장
85 await SaveSessionAsync(new UserSession
86 {
87 Provider = provider,
88 UserId = userId,
89 SessionId = sessionId,
90 IpAddress = ipAddress,
91 CreatedAt = DateTime.Now,
92 LastActivityAt = DateTime.Now,
93 IsActive = true,
94 ExpiryDate = rememberMe
95 ? DateTime.Now.AddDays(_jwtService.GetSettings.AutoLoginDays)
96 : DateTime.Now.AddDays(_jwtService.GetSettings.RefreshTokenDays),
97 });
98
99 // 중복 로그인 방지 설정이 활성화 된 경우
100 if (_jwtService.GetSettings.PreventDuplicateLogin)
101 {
102 await DeactivateOtherSessionAsync(userId, sessionId, "New Login");
103 }
104
105 // 캐시에 최신 세션 ID 저장
106 _activeSessionCache.AddOrUpdate(userKey, sessionId, (key, oldValue) => sessionId);
107 _cache.Set(userSessionKey, sessionId, _sessionCacheExpiry);
108
109 return sessionId;
110 }

118 public async Task<bool> ValidateSessionAsync(string userId, string provider, string sessionId)
119 {
120 string userKey = GetUserKey(userId, provider);
121 string userSessionKey = GetUserSessionKey(userId, provider);
122
123 // 캐시 확인
124 if (_activeSessionCache.TryGetValue(userKey, out var activeCachedSessionId))
125 {
126 return (activeCachedSessionId == sessionId);
127 }
128
129 if (_cache.TryGetValue(userSessionKey, out string cachedSessionId))
130 {
131 _activeSessionCache.TryAdd(userKey, cachedSessionId);
132 return (cachedSessionId == sessionId);
133 }
134
135 // DB 확인
136 var latestSession = await GetLatestSessionAsync(provider, userId);
137 if (latestSession == null || latestSession.SessionId != sessionId)
138 {
139 return false;
140 }
141
142 // 캐시 업데이트
143 _activeSessionCache.AddOrUpdate(userKey, latestSession.SessionId, (key, oldValue) => latestSession.SessionId);
144 _cache.Set(userSessionKey, latestSession.SessionId, _sessionCacheExpiry);
145
146 return (sessionId == latestSession.SessionId);
147 }

156 public async Task UpdateSessionActivityAsync(string userId, string provider, string sessionId)
157 {
158 // 세션 타임아웃이 설정되지 않은 경우 무시
159 if (!_jwtService.GetSettings.InactivityTimeout)
160 return;
161
162 string sessionCacheKey = GetSessionCacheKey(sessionId);
163
164 // 캐시에서 세션 찾기
165 if (!_cache.TryGetValue(sessionCacheKey, out UserSession session))
166 {
167 // DB에서 세션 조회
168 session = await GetSessionByIdAsync(sessionId);
169 if (session == null)
170 return;
171
172 // 캐시에 세션 저장
173 _cache.Set(sessionCacheKey, session, _sessionCacheExpiry);
174 }
175
176 // 마지막 Activity 시간 업데이트
177 session.LastActivityAt = DateTime.Now;
178
179 // DB의 세션 업데이트
180 _ = Task.Run(async () =>
181 {
182 await UpdateSessionAsync(session);
183 });
184 }

192 public async Task DeactivateSessionAsync(string sessionId, string reason = "Logout")
193 {
194 var session = await GetSessionByIdAsync(sessionId);
195 if (session == null || !session.IsActive)
196 return;
197
198 session.IsActive = false;
199 session.DeactivatedAt = DateTime.Now;
200 session.DeactivatedReason = reason;
201
202 // DB 업데이트
203 await UpdateSessionAsync(session);
204
205 // 캐시에서 세션 삭제
206 string sessionCacheKey = GetSessionCacheKey(sessionId);
207 string userKey = GetUserKey(session.UserId, session.Provider);
208
209 string userSessionKey = GetUserSessionKey(session.UserId, session.Provider);
210
211 _cache.Remove(sessionCacheKey);
212 _activeSessionCache.TryRemove(userKey, out _);
213 _cache.Remove(userSessionKey);
214 }

242 public async Task<bool> IsSessionInactiveAsync(string sessionId)
243 {
244 // 세션 타임아웃이 설정되지 않은 경우 무시
245 if (!_jwtService.GetSettings.InactivityTimeout)
246 return false;
247
248 var session = await GetSessionByIdAsync(sessionId);
249 if (session == null || !session.IsActive)
250 return true;
251
252 // 사용자의 세션 설정 가져오기
253 var user = await GetUserByIdAsync(session.UserId, session.Provider);
254 if (user == null)
255 {
256 return true;
257 }
258
259 if (!user.EnableSessionTimeout)
260 {
261 return false; // 세션 타임아웃 비활성화
262 }
263
264 // 마지막 활동 시간 확인
265 var inactiveTime = DateTime.Now - session.LastActivityAt;
266 var timeoutMinutes = user.SessionTimeoutMinutes == 0 ? _jwtService.GetSettings.InactivityTimeoutMinutes : user.SessionTimeoutMinutes;
267
268 return (inactiveTime.TotalMinutes > timeoutMinutes);
269 }

400 private async Task CleanupCacheAsync()
401 {
402 try
403 {
404 var activeSessions = await GetActiveSessionsAsync();
405 var validKeys = new HashSet<string>();
406
407 foreach (var session in activeSessions) // DB에서 유효한 세션 모두 가져오기
408 {
409 var userKey = GetUserKey(session.UserId, session.Provider);
410 validKeys.Add(userKey);
411 }
412
413 foreach (var key in _activeSessionCache.Keys)
414 {
415 if (!validKeys.Contains(key)) // DB에 존재하지 않는 세션 지우기
416 {
417 _activeSessionCache.TryRemove(key, out _);
418 }
419 }
420
421 await CleanupExpiredSessionsAsync(); // 만료된 세션 DB에서 정리(삭제)
422 }
423 catch (Exception e)
424 {
425 }
426 }
