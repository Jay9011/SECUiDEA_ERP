29 public override void OnActionExecuting(ActionExecutingContext context)
30 {
31 if (!context.HttpContext.Request.Headers.TryGetValue("X-API-KEY", out var apiKeyToken))
32 {
33 context.Result = new UnauthorizedObjectResult("API 키가 필요합니다.");
34 return;
35 }
36
37 var jwtService = context.HttpContext.RequestServices.GetRequiredService<JwtService>();
38 if (!jwtService.ValidateApiKeyToken(apiKeyToken, _issuer, out var principal))
39 {
40 context.Result = new UnauthorizedObjectResult("유효하지 않은 API 키입니다.");
41 return;
42 }
43
44 // 필수 Claim 검증
45 if (_requiredClaims != null)
46 {
47 foreach (var requiredClaim in _requiredClaims)
48 {
49 var claimValue = principal.FindFirstValue(requiredClaim.Key);
50 if (string.IsNullOrEmpty(claimValue) || claimValue != requiredClaim.Value)
51 {
52 context.Result = new UnauthorizedObjectResult($"필수 권한이 부족합니다: {requiredClaim.Key}");
53 return;
54 }
55 }
56 }
57
58 // API 키 ID와 서비스명 추출
59 var apiKeyId = principal.FindFirstValue(ClaimTypes.Authentication);
60 var service = principal.FindFirstValue(ClaimTypes.AuthenticationMethod);
61
62 // 요청 컨텍스트에 API 키 정보 추가
63 context.HttpContext.Items[ClaimTypes.Authentication] = apiKeyId;
64 context.HttpContext.Items[ClaimTypes.AuthenticationMethod] = service;
65
66 // 모든 추가 Claim을 HttpContext에 저장
67 foreach (var claim in principal.Claims)
68 {
69 // 이미 저장된 기본 Claim들은 제외
70 if (claim.Type != ClaimTypes.Authentication && claim.Type != ClaimTypes.AuthenticationMethod)
71 {
72 context.HttpContext.Items[$"ApiClaim_{claim.Type}"] = claim.Value;
73 }
74 }
75
76 base.OnActionExecuting(context);
77 }
