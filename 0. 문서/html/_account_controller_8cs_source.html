145 public string GenerateRandomToken(int length)
146 {
147 using var rng = System.Security.Cryptography.RandomNumberGenerator.Create();
148 var bytes = new byte[length];
149 rng.GetBytes(bytes);
150
151 // 랜덤 바이트를 Base32 문자로 변환
152 return new string(bytes.Select(b => BASE32_CHARS[b % BASE32_CHARS.Length]).ToArray());
153 }
