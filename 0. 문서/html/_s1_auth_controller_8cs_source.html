71 private async Task<IActionResult> ProcessLoginAsync(S1UserDTO request, string processedPassword)
72 {
73 try
74 {
75 // S1ACCESS 인증 처리
76 var user = await _userRepository.AuthenticateAsync(request.Id, processedPassword);
77 if (user == null)
78 {
79 return Unauthorized(BoolResultModel.Fail("인증 실패: 사용자 정보가 일치하지 않습니다."));
80 }
81
82 // 인증 성공 후 토큰 발급 및 세션 생성
83 string ipAddress = GetClientIpAddress();
84 var tokenResponse = await _authService.CompleteLoginAsync(
85 _userRepository.ProviderName,
86 user,
87 ipAddress,
88 request.rememberMe);
89
90 return Ok(BoolResultModel.Success("",
91 new Dictionary<string, object>
92 {
93 { "token", tokenResponse }
94 }));
95 }
96 catch (Exception ex)
97 {
98 return StatusCode(500, BoolResultModel.Fail("서버 오류가 발생했습니다."));
99 }
100 }
