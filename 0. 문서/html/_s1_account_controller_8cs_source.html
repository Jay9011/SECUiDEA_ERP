134 [Authorize]
135 [HttpPost]
136 public async Task<IActionResult> UpdatePassword([FromBody] S1UpdatePasswordModel model)
137 {
138 try
139 {
140 #region 인증 인가
141
142 // 로그인 사용자 ID 가져오기
143 var userId = User.FindFirstValue(ClaimTypes.NameIdentifier);
144
145 // 로그인 사용자 Role 가져오기
146 var userRoleClaimValue = User.FindFirstValue(ClaimTypes.Role);
147
148 // 두 정보 중 하나라도 없으면 인증 실패
149 if (string.IsNullOrEmpty(userId)
150 || string.IsNullOrEmpty(userRoleClaimValue))
151 {
152 return Ok(BoolResultModel.Fail("invalidUserAuthentication"));
153 }
154
155 #endregion
156
157 // 들어온 비밀번호 암호화
158 var shaEncryptPassword = _s1Sha512.Encrypt(model.Password);
159 // 현재 비밀번호 암호화
160 var currentPasswordHash = _s1Sha512.Encrypt(model.CurrentPassword);
161
162 #region 현재 비밀번호 검증
163
164 // 현재 비밀번호가 제공되지 않은 경우
165 if (string.IsNullOrEmpty(model.CurrentPassword))
166 {
167 return Ok(BoolResultModel.Fail("currentPasswordRequired"));
168 }
169
170 // 사용자 정보 가져오기
171 var repository = _userRepositoryFactory.GetRepository(StringClass.S1Access);
172 User user = await repository.GetUserModelByIdAsync(userId);
173
174 // 사용자 정보가 없는 경우
175 if (user == null)
176 {
177 return Unauthorized(BoolResultModel.Fail("userNotFound"));
178 }
179
180 // 현재 비밀번호 검증
181 if (user.Password != currentPasswordHash)
182 {
183 return Ok(BoolResultModel.Fail("currentPasswordIncorrect"));
184 }
185
186 #endregion
187
188 // 비밀번호 변경 파라미터 설정
189 S1AccountPassRegParam param = new S1AccountPassRegParam
190 {
191 Type = "change",
192 Role = userRoleClaimValue,
193 ID = userId,
194 Password = shaEncryptPassword,
195 UpdateIP = GetClientIpAddress()
196 };
197
198 // 비밀번호 설정
199 var result = await _s1Setup.DAL.ExecuteProcedureAsync(_s1Setup, "SECUiDEA_PasswordREG", param);
200 if (result.IsSuccess
201 && result.ReturnValue == 1)
202 {
203 return Ok(BoolResultModel.Success("passwordUpdateSuccess"));
204 }
205 }
