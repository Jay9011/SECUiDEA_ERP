39 public async Task<IActionResult> SendRequest([FromBody] SendRequestModel request)
40 {
41 try
42 {
43 // API 키 인증
44 var apiKeyId = HttpContext.Items[ClaimTypes.Authentication] as string;
45 var serviceName = HttpContext.Items[ClaimTypes.AuthenticationMethod] as string;
46
47 AligoAcountDTO aligoAcountDto = await GetAligoAccountInfo(_secuidea);
48 AligoTemplateDTO aligoTemplateDto = await GetAligoTemplateInfo(_secuidea, request.Gubun);
49
50 // 만약 둘 중 하나가 null이라면 에러 처리
51 if (aligoAcountDto == null || aligoTemplateDto == null)
52 {
53 return BadRequest("Aligo account or template information is missing.");
54 }
55
56 IAligoClient aligoClient = AligoClientFactory.Create(aligoAcountDto.APIKey, aligoAcountDto.UserID, aligoTemplateDto.SenderKey);
57
58 // AligoTemplateDTO의 Path가 null 혹은 빈 값이 아니라면 도메인 + Path를 사용
59 string domain = aligoAcountDto.Url;
60 if (!string.IsNullOrEmpty(aligoTemplateDto.Path))
61 {
62 domain = $"{domain}{aligoTemplateDto.Path}";
63 }
64
65 // 템플릿 정보 조회
66 var templateResult = await aligoClient.TemplateService.GetTemplatesAsync(aligoTemplateDto.TPL_CODE);
67 if (templateResult.Success && templateResult.Templates.Count > 0)
68 {
69 var template = templateResult.Templates[0]; // 일단 템플릿은 하나만 사용한다고 가정
70 var tplContent = template.Content;
71 var tplName = template.Name;
72
73 // 수신자별로 메시지 내용과 변수 세팅
74 foreach (var receiver in request.ReceiverList)
75 {
76 receiver.Subject = tplName;
77 receiver.TemplateContent = tplContent;
78 receiver.FailoverSubject = tplName;
79 receiver.FailoverMessage = tplContent;
80
81 // variables는 클라이언트 측에서 세팅
82
83 // queryVariables가 1개 이상이라면 해당 값에 도메인 추가 (queryVariables도 클라이언트 측에서 세팅)
84 if (receiver.QueryVariables != null && receiver.QueryVariables.Count > 0)
85 {
86 foreach (var queryVariable in receiver.QueryVariables)
87 {
88 var key = queryVariable.Key;
89 var value = $"{domain}{queryVariable.Value}";
90
91 // 도메인 추가
92 receiver.QueryVariables[key] = value;
93 }
94 }
95 }
96
97 // 메시지 전송 요청 생성
98 var sendRequest = TemplateHelper.CreateSendRequest(
99 templateCode: aligoTemplateDto.TPL_CODE,
100 sender: aligoAcountDto.SenderPhone,
101 receiverList: request.ReceiverList
102 );
103
104 // 메시지 전송
105 aligoClient.MessageService.SendMessageAsync(sendRequest);
106 }
107
108 return Ok(BoolResultModel.Success("Success to send message"));
109 }
110 catch (Exception e)
111 {
112 return BadRequest(BoolResultModel.Fail(e.Message));
113 }
114
115 return BadRequest(BoolResultModel.Fail("Failed to send message"));
116 }
